> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 5MB Korean morphological analyzer that runs in web browsers via WASM, with `Garu.load()`, `analyze()`, and `tokenize()` APIs.

**Architecture:** Rust core compiled to WASM via wasm-pack. Custom BiLSTM+CRF inference engine (no ONNX Runtime). Jamo-level input encoding (~200 vocab) to minimize embedding size. LOUDS Trie compressed dictionary for fast lookup. JS/TS wrapper exposes the public API. Python training pipeline (PyTorch) produces model weights exported to custom binary format with INT8 quantization.

**Tech Stack:** Rust, wasm-pack, wasm-bindgen, TypeScript, PyTorch, Vitest

**Spec:** `docs/superpowers/specs/2026-03-24-kiwi-lite-design.md`

Korean Unicode block (0xAC00-0xD7A3) decomposes into lead + vowel + tail. This is the foundation for the Jamo-level input encoding the BiLSTM expects.

- [ ] **Step 1: Write Jamo tests and implementation**

`crates/garu-core/src/jamo.rs`:

```rust
/// Jamo decomposition and encoding for Korean text.
/// Converts Korean syllables into individual Jamo (consonant/vowel) sequences.
/// Non-Korean characters pass through as special tokens.

const HANGUL_BASE: u32 = 0xAC00;
const HANGUL_END: u32 = 0xD7A3;
const VOWEL_COUNT: u32 = 21;
const TAIL_COUNT: u32 = 28; // includes "no tail" = 0

const LEADS: [char; 19] = [
    'ㄱ','ㄲ','ㄴ','ㄷ','ㄸ','ㄹ','ㅁ','ㅂ','ㅃ','ㅅ',
    'ㅆ','ㅇ','ㅈ','ㅉ','ㅊ','ㅋ','ㅌ','ㅍ','ㅎ',
];

const VOWELS: [char; 21] = [
    'ㅏ','ㅐ','ㅑ','ㅒ','ㅓ','ㅔ','ㅕ','ㅖ','ㅗ','ㅘ',
    'ㅙ','ㅚ','ㅛ','ㅜ','ㅝ','ㅞ','ㅟ','ㅠ','ㅡ','ㅢ','ㅣ',
];

const TAILS: [Option<char>; 28] = [
    None,
    Some('ㄱ'),Some('ㄲ'),Some('ㄳ'),Some('ㄴ'),Some('ㄵ'),Some('ㄶ'),
    Some('ㄷ'),Some('ㄹ'),Some('ㄺ'),Some('ㄻ'),Some('ㄼ'),Some('ㄽ'),
    Some('ㄾ'),Some('ㄿ'),Some('ㅀ'),Some('ㅁ'),Some('ㅂ'),Some('ㅄ'),
    Some('ㅅ'),Some('ㅆ'),Some('ㅇ'),Some('ㅈ'),Some('ㅊ'),Some('ㅋ'),
    Some('ㅌ'),Some('ㅍ'),Some('ㅎ'),
];

/// Decompose a single Hangul syllable into Jamo characters.
pub fn decompose(ch: char) -> Option<(char, char, Option<char>)> {
    let code = ch as u32;
    if code < HANGUL_BASE || code > HANGUL_END {
        return None;
    }
    let offset = code - HANGUL_BASE;
    let lead_idx = offset / (VOWEL_COUNT * TAIL_COUNT);
    let vowel_idx = (offset % (VOWEL_COUNT * TAIL_COUNT)) / TAIL_COUNT;
    let tail_idx = offset % TAIL_COUNT;

    Some((
        LEADS[lead_idx as usize],
        VOWELS[vowel_idx as usize],
        TAILS[tail_idx as usize],
    ))
}

/// Special token IDs for the encoder
pub const PAD_ID: u16 = 0;
pub const UNK_ID: u16 = 1;
pub const SPACE_ID: u16 = 2;
pub const NUM_ID: u16 = 3;
pub const LATIN_ID: u16 = 4;
pub const PUNCT_ID: u16 = 5;
pub const JAMO_OFFSET: u16 = 6;

/// Encode a Jamo character to its vocab ID.
fn jamo_to_id(ch: char) -> u16 {
    let code = ch as u32;
    if (0x3131..=0x3163).contains(&code) {
        JAMO_OFFSET + (code - 0x3131) as u16
    } else {
        UNK_ID
    }
}

/// Encode a string into a sequence of vocab IDs for the BiLSTM.
pub fn encode(text: &str) -> Vec<u16> {
    let mut ids = Vec::with_capacity(text.len() * 3);
    for ch in text.chars() {
        if let Some((lead, vowel, tail)) = decompose(ch) {
            ids.push(jamo_to_id(lead));
            ids.push(jamo_to_id(vowel));
            if let Some(t) = tail {
                ids.push(jamo_to_id(t));
            }
        } else if ch == ' ' {
            ids.push(SPACE_ID);
        } else if ch.is_ascii_digit() {
            ids.push(NUM_ID);
        } else if ch.is_ascii_alphabetic() {
            ids.push(LATIN_ID);
        } else if ch.is_ascii_punctuation() {
            ids.push(PUNCT_ID);
        } else {
            ids.push(UNK_ID);
        }
    }
    ids
}

/// Total vocab size (for embedding table).
pub const VOCAB_SIZE: usize = 6 + 51; // 6 special + 51 Jamo (0x3131..0x3163)
```

- [ ] **Step 1: Write Trie tests and implementation**

`crates/garu-core/src/trie.rs`:
```rust
use crate::types::Pos;

#[derive(Debug, Clone)]
pub struct Morpheme {
    pub text: String,
    pub pos: Pos,
}

#[derive(Debug, Clone)]
pub struct DictEntry {
    pub morphemes: Vec<Morpheme>,
    pub score: f32,
}

pub struct Dict {
    nodes: Vec<TrieNode>,
}

struct TrieNode {
    children: Vec<(char, usize)>,
    entries: Vec<DictEntry>,
}

impl Dict {
    pub fn new() -> Self {
        Dict {
            nodes: vec![TrieNode { children: Vec::new(), entries: Vec::new() }],
        }
    }

    pub fn insert(&mut self, word: &str, entry: DictEntry) {
        let mut node_idx = 0;
        for ch in word.chars() {
            let next = self.nodes[node_idx]
                .children
                .iter()
                .find(|(c, _)| *c == ch)
                .map(|(_, idx)| *idx);

            node_idx = match next {
                Some(idx) => idx,
                None => {
                    let new_idx = self.nodes.len();
                    self.nodes.push(TrieNode { children: Vec::new(), entries: Vec::new() });
                    self.nodes[node_idx].children.push((ch, new_idx));
                    new_idx
                }
            };
        }
        self.nodes[node_idx].entries.push(entry);
    }

    pub fn lookup(&self, word: &str) -> &[DictEntry] {
        let mut node_idx = 0;
        for ch in word.chars() {
            match self.nodes[node_idx].children.iter().find(|(c, _)| *c == ch) {
                Some((_, idx)) => node_idx = *idx,
                None => return &[],
            }
        }
        &self.nodes[node_idx].entries
    }

    pub fn common_prefix_search(&self, text: &str) -> Vec<(usize, &[DictEntry])> {
        let mut results = Vec::new();
        let mut node_idx = 0;
        for (i, ch) in text.chars().enumerate() {
            match self.nodes[node_idx].children.iter().find(|(c, _)| *c == ch) {
                Some((_, idx)) => {
                    node_idx = *idx;
                    if !self.nodes[node_idx].entries.is_empty() {
                        results.push((i + 1, self.nodes[node_idx].entries.as_slice()));
                    }
                }
                None => break,
            }
        }
        results
    }

    pub fn len(&self) -> usize {
        self.nodes.iter().map(|n| n.entries.len()).sum()
    }

    pub fn is_empty(&self) -> bool {
        self.len() == 0
    }

    pub fn to_bytes(&self) -> Vec<u8> {
        let mut buf = Vec::new();
        buf.extend_from_slice(b"GARU");
        buf.extend_from_slice(&1u32.to_le_bytes());
        buf.extend_from_slice(&(self.len() as u32).to_le_bytes());
        self.serialize_entries(&mut buf, 0, String::new());
        buf
    }

    fn serialize_entries(&self, buf: &mut Vec<u8>, node_idx: usize, prefix: String) {
        let node = &self.nodes[node_idx];
        if !node.entries.is_empty() {
            let word_bytes = prefix.as_bytes();
            buf.extend_from_slice(&(word_bytes.len() as u16).to_le_bytes());
            buf.extend_from_slice(word_bytes);
            buf.push(node.entries.len() as u8);
            for entry in &node.entries {
                buf.push(entry.morphemes.len() as u8);
                for m in &entry.morphemes {
                    let text_bytes = m.text.as_bytes();
                    buf.extend_from_slice(&(text_bytes.len() as u16).to_le_bytes());
                    buf.extend_from_slice(text_bytes);
                    buf.push(m.pos as u8);
                }
                buf.extend_from_slice(&entry.score.to_le_bytes());
            }
        }
        for (ch, child_idx) in &node.children {
            let mut child_prefix = prefix.clone();
            child_prefix.push(*ch);
            self.serialize_entries(buf, *child_idx, child_prefix);
        }
    }

    pub fn from_bytes(data: &[u8]) -> Result<Self, String> {
        if data.len() < 12 || &data[0..4] != b"GARU" {
            return Err("Invalid dictionary format".to_string());
        }
        let version = u32::from_le_bytes(data[4..8].try_into().unwrap());
        if version != 1 {
            return Err(format!("Unsupported version: {version}"));
        }
        let entry_count = u32::from_le_bytes(data[8..12].try_into().unwrap()) as usize;

        let mut dict = Dict::new();
        let mut cursor = 12;
        let mut loaded = 0;

        while cursor < data.len() && loaded < entry_count {
            let word_len = u16::from_le_bytes(data[cursor..cursor+2].try_into().unwrap()) as usize;
            cursor += 2;
            let word = std::str::from_utf8(&data[cursor..cursor+word_len])
                .map_err(|e| format!("Invalid UTF-8: {e}"))?
                .to_string();
            cursor += word_len;

            let num_entries = data[cursor] as usize;
            cursor += 1;

            for _ in 0..num_entries {
                let num_morphemes = data[cursor] as usize;
                cursor += 1;
                let mut morphemes = Vec::with_capacity(num_morphemes);
                for _ in 0..num_morphemes {
                    let text_len = u16::from_le_bytes(data[cursor..cursor+2].try_into().unwrap()) as usize;
                    cursor += 2;
                    let text = std::str::from_utf8(&data[cursor..cursor+text_len])
                        .map_err(|e| format!("Invalid UTF-8: {e}"))?
                        .to_string();
                    cursor += text_len;
                    let pos_byte = data[cursor];
                    cursor += 1;
                    if pos_byte > 38 {
                        return Err(format!("Invalid POS byte: {pos_byte}"));
                    }
                    let pos: Pos = unsafe { std::mem::transmute(pos_byte) };
                    morphemes.push(Morpheme { text, pos });
                }
                let score = f32::from_le_bytes(data[cursor..cursor+4].try_into().unwrap());
                cursor += 4;
                dict.insert(&word, DictEntry { morphemes, score });
                loaded += 1;
            }
        }
        Ok(dict)
    }
}
```

```rust
/// BiLSTM forward pass with INT8 quantized weights.

#[derive(Debug, Clone)]
pub struct QuantizedMatrix {
    pub data: Vec<i8>,
    pub rows: usize,
    pub cols: usize,
    pub scale: f32,
}

impl QuantizedMatrix {
    pub fn matvec(&self, input: &[f32]) -> Vec<f32> {
        assert_eq!(input.len(), self.cols);
        let mut output = vec![0.0f32; self.rows];
        for r in 0..self.rows {
            let row_start = r * self.cols;
            let mut sum = 0.0f32;
            for c in 0..self.cols {
                sum += (self.data[row_start + c] as f32) * self.scale * input[c];
            }
            output[r] = sum;
        }
        output
    }
}

#[derive(Debug, Clone)]
pub struct Embedding {
    pub weights: Vec<f32>,
    pub vocab_size: usize,
    pub embed_dim: usize,
}

impl Embedding {
    pub fn lookup(&self, id: u16) -> &[f32] {
        let idx = id as usize;
        if idx >= self.vocab_size {
            return &self.weights[0..self.embed_dim];
        }
        let start = idx * self.embed_dim;
        &self.weights[start..start + self.embed_dim]
    }
}

#[derive(Debug, Clone)]
pub struct LstmLayer {
    pub w_ii: QuantizedMatrix,
    pub w_hi: QuantizedMatrix,
    pub b_i: Vec<f32>,
    pub w_if: QuantizedMatrix,
    pub w_hf: QuantizedMatrix,
    pub b_f: Vec<f32>,
    pub w_ig: QuantizedMatrix,
    pub w_hg: QuantizedMatrix,
    pub b_g: Vec<f32>,
    pub w_io: QuantizedMatrix,
    pub w_ho: QuantizedMatrix,
    pub b_o: Vec<f32>,
    pub hidden_size: usize,
}

impl LstmLayer {
    pub fn step(&self, input: &[f32], h: &[f32], c: &[f32]) -> (Vec<f32>, Vec<f32>) {
        let hs = self.hidden_size;
        let i_gate = sigmoid(&add3(&self.w_ii.matvec(input), &self.w_hi.matvec(h), &self.b_i));
        let f_gate = sigmoid(&add3(&self.w_if.matvec(input), &self.w_hf.matvec(h), &self.b_f));
        let g_gate = tanh_vec(&add3(&self.w_ig.matvec(input), &self.w_hg.matvec(h), &self.b_g));
        let o_gate = sigmoid(&add3(&self.w_io.matvec(input), &self.w_ho.matvec(h), &self.b_o));

        let mut new_c = vec![0.0; hs];
        let mut new_h = vec![0.0; hs];
        for j in 0..hs {
            new_c[j] = f_gate[j] * c[j] + i_gate[j] * g_gate[j];
            new_h[j] = o_gate[j] * new_c[j].tanh();
        }
        (new_h, new_c)
    }
}

#[derive(Debug, Clone)]
pub struct BiLstm {
    pub layers: Vec<(LstmLayer, LstmLayer)>,
    pub hidden_size: usize,
}

impl BiLstm {
    pub fn forward(&self, embeddings: &[Vec<f32>]) -> Vec<Vec<f32>> {
        let seq_len = embeddings.len();
        let hs = self.hidden_size;
        let mut layer_input = embeddings.to_vec();

        for (fwd, bwd) in &self.layers {
            let mut fwd_outputs = Vec::with_capacity(seq_len);
            let mut h = vec![0.0; hs];
            let mut c = vec![0.0; hs];
            for t in 0..seq_len {
                let (new_h, new_c) = fwd.step(&layer_input[t], &h, &c);
                fwd_outputs.push(new_h.clone());
                h = new_h;
                c = new_c;
            }

            let mut bwd_outputs = vec![vec![0.0; hs]; seq_len];
            h = vec![0.0; hs];
            c = vec![0.0; hs];
            for t in (0..seq_len).rev() {
                let (new_h, new_c) = bwd.step(&layer_input[t], &h, &c);
                bwd_outputs[t] = new_h.clone();
                h = new_h;
                c = new_c;
            }

            layer_input = (0..seq_len)
                .map(|t| {
                    let mut concat = fwd_outputs[t].clone();
                    concat.extend_from_slice(&bwd_outputs[t]);
                    concat
                })
                .collect();
        }

        layer_input
    }
}

fn sigmoid(v: &[f32]) -> Vec<f32> {
    v.iter().map(|&x| 1.0 / (1.0 + (-x).exp())).collect()
}

fn tanh_vec(v: &[f32]) -> Vec<f32> {
    v.iter().map(|&x| x.tanh()).collect()
}

fn add3(a: &[f32], b: &[f32], c: &[f32]) -> Vec<f32> {
    a.iter().zip(b.iter()).zip(c.iter()).map(|((&a, &b), &c)| a + b + c).collect()
}
```

```rust
#[derive(Debug, Clone)]
pub struct Crf {
    pub transitions: Vec<Vec<f32>>,
    pub num_tags: usize,
}

impl Crf {
    pub fn decode(&self, emissions: &[Vec<f32>]) -> (Vec<usize>, f32) {
        let seq_len = emissions.len();
        if seq_len == 0 {
            return (vec![], 0.0);
        }

        let n = self.num_tags;
        let mut viterbi = vec![vec![f32::NEG_INFINITY; n]; seq_len];
        let mut backpointers = vec![vec![0usize; n]; seq_len];

        for j in 0..n {
            viterbi[0][j] = emissions[0][j];
        }

        for t in 1..seq_len {
            for j in 0..n {
                let mut best_score = f32::NEG_INFINITY;
                let mut best_prev = 0;
                for i in 0..n {
                    let score = viterbi[t - 1][i] + self.transitions[i][j] + emissions[t][j];
                    if score > best_score {
                        best_score = score;
                        best_prev = i;
                    }
                }
                viterbi[t][j] = best_score;
                backpointers[t][j] = best_prev;
            }
        }

        let mut best_tag = 0;
        let mut best_score = f32::NEG_INFINITY;
        for j in 0..n {
            if viterbi[seq_len - 1][j] > best_score {
                best_score = viterbi[seq_len - 1][j];
                best_tag = j;
            }
        }

        let mut tags = vec![0usize; seq_len];
        tags[seq_len - 1] = best_tag;
        for t in (0..seq_len - 1).rev() {
            tags[t] = backpointers[t + 1][tags[t + 1]];
        }

        (tags, best_score)
    }

    pub fn decode_topn(&self, emissions: &[Vec<f32>], n: usize) -> Vec<(Vec<usize>, f32)> {
        if n == 1 {
            return vec![self.decode(emissions)];
        }
        let seq_len = emissions.len();
        if seq_len == 0 {
            return vec![(vec![], 0.0)];
        }
        let num_tags = self.num_tags;
        let mut beam: Vec<(f32, Vec<usize>)> = (0..num_tags)
            .map(|j| (emissions[0][j], vec![j]))
            .collect();

        for t in 1..seq_len {
            let mut candidates = Vec::new();
            for (prev_score, path) in &beam {
                let prev_tag = *path.last().unwrap();
                for j in 0..num_tags {
                    let score = prev_score + self.transitions[prev_tag][j] + emissions[t][j];
                    let mut new_path = path.clone();
                    new_path.push(j);
                    candidates.push((score, new_path));
                }
            }
            candidates.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());
            candidates.truncate(n);
            beam = candidates;
        }
        beam
    }
}
```

```rust
pub struct GaruWasm { analyzer: Analyzer }

#[wasm_bindgen]
impl GaruWasm {
    #[wasm_bindgen(constructor)]

    }
```
